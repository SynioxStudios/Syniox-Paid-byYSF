local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer

local config = {
    whitelist = "https://pastefy.app/B1jo6EO6/raw",
    script = "https://api.jnkie.com/api/v1/luascripts/public/0250fcddd34be3e7d55626c3b286cd0d1e35f41f9faecbecf233cb87f4e19ab2/download",

}

if typeof(loadstring) ~= "function" or type(game.HttpGet) ~= "function" then
    warn("Execution not supported!")
    return
end

local function StartLoader()
    local success, content = pcall(function() return game:HttpGet(config.whitelist) end)
    
    if success then
        local loadSuccess, whitelist = pcall(function() return loadstring(content)() end)
        
        if loadSuccess and type(whitelist) == "table" then
            local isAllowed = false
            for name, _ in pairs(whitelist) do
                if name:lower() == LocalPlayer.Name:lower() then
                    isAllowed = true
                    break
                end
            end

            if isAllowed then
                local mainScriptSuccess, mainScriptSource = pcall(function() 
                    return game:HttpGet(config.script) 
                end)
                
                if mainScriptSuccess and mainScriptSource ~= "" then
                    loadstring(mainScriptSource)()
                else
                    warn("Failed to load main script.")
                end
            else
                LocalPlayer:Kick("Not Whitelisted. Where did you get this? Your account is mine now! 💀")
            end
        else
            LocalPlayer:Kick("System Error: Invalid Data")
        end
    else
        LocalPlayer:Kick("Network Error: Connection Failed")
    end
end

StartLoader()
